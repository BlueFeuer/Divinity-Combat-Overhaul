// Arbiter timer
IF
GameStarted(_Level, _)
THEN
IterateParties("Blue_ArbiterTrigger2");
//Arbiter Timer handles player stats, and in combat triggers extra checks to the current turn holder.
TimerCancel("Blue_ArbiterTimer");
TimerLaunch("Blue_ArbiterTimer", 250);
//Combat Timer triggers checks to every character currently in combat.
TimerCancel("Blue_ArbiterCombatTimer");
TimerLaunch("Blue_ArbiterCombatTimer", 300);

IF
TimerFinished("Blue_ArbiterTimer")
THEN
IterateParties("Blue_ArbiterTrigger");
TimerLaunch("Blue_ArbiterTimer", 250);

IF
TimerFinished("Blue_ArbiterTimer")
AND
DB_CombatCharacters(_Character, _Int)
THEN
SetStoryEvent(_Character, "Blue_ArbiterTurnTrigger");

IF
TimerFinished("Blue_ArbiterCombatTimer")
THEN
TimerLaunch("Blue_ArbiterCombatTimer", 300);

// Combat event should trigger roughly every 4 seconds and when forced via a tag
QRY
QRY_Blue_GetCombatCheck((CHARACTERGUID)_Character)
AND
IsTagged(_Character, "Blue_ForceEnableArbiter", 1)
THEN
DB_NOOP(1);

QRY
QRY_Blue_GetCombatCheck((CHARACTERGUID)_Character)
AND
Random(100, _Rand)
AND
_Rand < 7
THEN
DB_NOOP(1);

IF
TimerFinished("Blue_ArbiterCombatTimer")
AND
DB_CombatCharacters(_Character, _Int)
AND
QRY_Blue_GetCombatCheck(_Character)
THEN
SetStoryEvent(_Character, "Blue_ArbiterCombatTrigger");
